## Install

```bash
npm install --save react-pull-from-edge
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'react-pull-from-edge'
import 'react-pull-from-edge/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
